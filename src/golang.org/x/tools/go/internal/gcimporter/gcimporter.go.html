// Type =
//	BasicType | TypeName | ArrayType | SliceType | StructType |
//      PointerType | FuncType | InterfaceType | MapType | ChanType |
//      "(" Type ")" .
//
// BasicType   = ident .
// TypeName    = ExportedName .
// SliceType   = "[" "]" Type .
// PointerType = "*" Type .
// FuncType    = "func" Signature .
//
func ( *parser) ( *types.Package) types.Type {
	switch .tok {
	case scanner.Ident:
		switch .lit {
		default:
			return .parseBasicType()
		case "struct":
			return .parseStructType()
		case "func":
			// FuncType
			.next()
			return .parseSignature(nil)
		case "interface":
			return .parseInterfaceType()
		case "map":
			return .parseMapType()
		case "chan":
			return .parseChanType()
		}
	case '@':
		// TypeName
		,  := .parseExportedName()
		return declTypeName(, ).Type()
	case '[':
		.next() // look ahead
		if .tok == ']' {
			// SliceType
			.next()
			return types.NewSlice(.())
		}
		return .parseArrayType()
	case '*':
		// PointerType
		.next()
		return types.NewPointer(.())
	case '<':
		return .parseChanType()
	case '(':
		// "(" Type ")"
		.next()
		 := .()
		.expect(')')
		return 
	}
	.errorf("expected type, got %s (%q)", scanner.TokenString(.tok), .lit)
	return nil
}

// ConstDecl   = "const" ExportedName [ Type ] "=" Literal .
// Literal     = bool_lit | int_lit | float_lit | complex_lit | rune_lit | string_lit .
// bool_lit    = "true" | "false" .
// complex_lit = "(" float_lit "+" float_lit "i" ")" .
// rune_lit    = "(" int_lit "+" int_lit ")" .
// string_lit  = `"` { unicode_char } `"` .
//

		 = .parseType(nil)
	}

	var  types.Type
	var  constant.Value

	case scanner.Ident:
		// bool_lit
		if .lit != "true" && .lit != "false" {
			.error("expected true or false")
		}
		 = types.Typ[types.UntypedBool]
		 = constant.MakeBool(.lit == "true")
		.next()

	case '-', scanner.Int:
		// int_lit
		,  = .parseNumber()

	case '(':
		// complex_lit or rune_lit
		.next()
		if .tok == scanner.Char {
			.next()
			.expect('+')
			 = types.Typ[types.UntypedRune]
			_,  = .parseNumber()
			.expect(')')
			break
		}
		,  := .parseNumber()
		.expect('+')
		,  := .parseNumber()
		.expectKeyword("i")
		.expect(')')
		 = types.Typ[types.UntypedComplex]
		 = constant.BinaryOp(, token.ADD, constant.MakeImag())

	case scanner.Char:
		// rune_lit
		 = types.Typ[types.UntypedRune]
		 = constant.MakeFromLiteral(.lit, token.CHAR, 0)
