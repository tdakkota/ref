// state tracks error printing state. It implements fmt.State.
type state struct {
	fmt.State
	buf bytes.Buffer

	printDetail bool
	inDetail    bool
	needColon   bool
	needNewline bool
}

func ( *state) ( []byte) ( int,  error) {

		if len() == 0 {
			return 0, nil

			.needNewline = true
			if [0] == '\n' {
				 = [1:]

					.buf.WriteByte(':')
					.needColon = false
				}
				.buf.Write(detailSep)
				.needNewline = false
			}
			if  == '\n' {
				.buf.Write([:])
				 =  + 1
				.needNewline = true
			}
		}
		.buf.Write([:])
		if !.inDetail {
			.needColon = true
		}
	} else if !.inDetail {
		.buf.Write()
	}
	return len(), nil
}

// printer wraps a state to implement an xerrors.Printer.
type printer state

func ( *printer) ( ...interface{}) {
	if !.inDetail || .printDetail {
		_, _ = fmt.Fprint((*state)(), ...)
	}
}

func ( *printer) ( string,  ...interface{}) {
	if !.inDetail || .printDetail {
		_, _ = fmt.Fprintf((*state)(), , ...)
	}
}

func ( *printer) () bool {
	.inDetail = true
	return .printDetail
